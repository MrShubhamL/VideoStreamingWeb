This is database structure

CREATE DATABASE employee;

CREATE TABLE emp_details (id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,empName VARCHAR(55) 
NOT NULL, empEmail VARCHAR(200) UNIQUE NOT NULL, empContact VARCHAR(10) NOT NULL UNIQUE, 
empDepartment VARCHAR(100), password VARCHAR(200));

CREATE TABLE videos (id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,video_name VARCHAR(55) 
NOT NULL, video_description VARCHAR(200) NOT NULL, video_upload_date VARCHAR(10) NOT NULL, 
video_url VARCHAR(500));
